# deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: wisecow-app
spec:
  replicas: 3  # Adjust as per your requirement
  selector:
    matchLabels:
      app: wisecow
  template:
    metadata:
      labels:
        app: wisecow
    spec:
      containers:
        - name: wisecow
          image: yourdockerhubusername/wisecow-app:latest  # Replace with your Docker Hub username and image name
          ports:
            - containerPort: 3000
